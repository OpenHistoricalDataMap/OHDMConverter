\newglossaryentry{psql}{
	name=psql,
	description={psql ist ein Terminal-basiertes Front-End für PostgreSQL. Es ermöglicht Ihnen, Abfragen interaktiv einzugeben, sie an PostgreSQL zu senden und die Abfrageergebnisse zu sehen. Alternativ kann die Eingabe auch aus einer Datei oder über Kommandozeilenargumente erfolgen. Darüber hinaus bietet psql eine Reihe von Meta-Befehlen und verschiedene Shell-ähnliche Funktionen, die das Schreiben von Skripten und die Automatisierung einer Vielzahl von Aufgaben erleichtern.(vgl. \cite{postgres-psql})}
}

\newglossaryentry{flexoutput}{
	name={Flex Output},
	description={Die flexible Ausgabe ermöglicht, wie der Name schon sagt, eine flexible Konfiguration, die osm2pgsql mitteilt, welche OSM-Daten in Ihrer Datenbank gespeichert werden sollen und wo und wie genau. Sie wird durch eine Lua-Datei konfiguriert, die
		\begin{itemize}
			\item die Struktur der Ausgabetabellen definiert und
			\item Funktionen definiert, um die OSM-Daten auf das Datenformat der Datenbank abzubilden
		\end{itemize}(vlg. \cite{osm2pgsql-flex})}
